def write_file(folder, name, content):  
    os.makedirs(folder, exist_ok=True)  
    path = os.path.join(folder, name)  
  
    with open(path, "w", encoding="utf-8") as f:  
        f.write(content)

## 📂 читаем логи

def read_logs():  
    logs_dir = "logs"  
    entries = []  
  
    if not os.path.exists(logs_dir):  
        return entries  
  
    for f in os.listdir(logs_dir):  
        path = os.path.join(logs_dir, f)  
        with open(path, encoding="utf-8") as file:  
            entries.append(file.read())  
  
    return entries

---

## 🧠 анализ логов

def analyze_logs():  
    logs = read_logs()  
  
    if not logs:  
        print("[NO LOGS]")  
        return  
  
    print("\n[ANALYSIS START]\n")  
  
    for log in logs:  
        if "нужно" in log.lower():  
            print("→ Найдено улучшение:", log.strip()[:100])  
  
        if "ошибка" in log.lower() or "не работает" in log.lower():  
            print("→ Найдена проблема:", log.strip()[:100])

---

## 🚀 генерация улучшений

def evolve():  
    logs = read_logs()  
  
    suggestions = []  
  
    for log in logs:  
        if "нужно" in log.lower():  
            suggestions.append("Создать новую node или улучшить систему")  
  
    if not suggestions:  
        print("[NO EVOLUTION NEEDED]")  
        return  
  
    print("\n[EVOLUTION SUGGESTIONS]:")  
    for s in suggestions:  
        print("-", s)

## 📂 работа с памятью

```python
import json
import datetime

MEMORY_DIR = "memory"

def load_json(filename, default):
    path = f"{MEMORY_DIR}/{filename}"
    if not os.path.exists(path):
        return default
    with open(path, encoding="utf-8") as f:
        return json.load(f)

def save_json(filename, data):
    os.makedirs(MEMORY_DIR, exist_ok=True)
    path = f"{MEMORY_DIR}/{filename}"
    with open(path, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, ensure_ascii=False)
```

## 🧠 загрузка памяти

```python
def load_memory():
    profile = load_json("profile.json", {
```

## 🧠 server.py

```python
from flask import Flask, request, jsonify
from agent import personalized_think, save_interaction, update_profile_ai

app = Flask(__name__)

@app.route("/ai", methods=["POST"])
def ai_endpoint():
    data = request.json
    query = data.get("query")

    result = personalized_think(query)

    save_interaction(query, result)
    update_profile_ai(query, result)

    return jsonify({"response": result})

app.run(port=5000)
```

---

## 🌐 index.html (UI)

```html
<!DOCTYPE html>
<html>
<head>
    <title>AI OS</title>
</head>
<body>

<h2>AI OS</h2>

<textarea id="input" placeholder="Напиши что хочешь..." rows="4"></textarea>
<br><br>

<button onclick="send()">Спросить</button>

<pre id="output"></pre>

<script>
async function send() {
    let query = document.getElementById("input").value;

    let res = await fetch("http://localhost:5000/ai", {
        method: "POST",
        headers: {"Content-Type": "application/json"},
        body: JSON.stringify({query})
    });

    let data = await res.json();
    document.getElementById("output").innerText = data.response;
}
</script>

</body>
</html>
```

## 🔥 1. Контекст node (твоя уникальность!)

Добавь в `personalized_think()`:

```python
def load_nodes_context():
    nodes = []
    for root, _, files in os.walk("nodes"):
        for f in files:
            if f.endswith(".md"):
                with open(os.path.join(root, f), encoding="utf-8") as file:
                    nodes.append(file.read())
    return "\n\n".join(nodes[:5])  # ограничим
```

---

И в prompt:

```python
nodes_context = load_nodes_context()

prompt = f"""
Ты — AI-ОС с архитектурой node.

Контекст системы:
{nodes_context}

Профиль:
{profile}

{patterns}

История:
{history[-5:]}

{query}
```
